        return n.Subject

var subject4k = _.mapValues(groups,function(n){ 
	var  classLvl = _.pluck(n,"CrsLvlNum")
	classes4k = _.filter(classLvl,function(n){
		return n == "4XXX"
	})
        return  classes4k.length
})
var sort4k = _.sortBy(subject4k, function(n,key){
	return n
})
